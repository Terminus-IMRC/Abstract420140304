\foilhead{5次魔方陣の定義}
\begin{itemize}
\item 5 $\times$ 5 のマス目
\item 1 $\sim$ 25 の数を1つずつ入れる
\item 縦の5列 $\cdot$ 横の5列 $\cdot$ 斜めの2列の合計が全て等しい
\begin{boldequation}
L=\frac{1}{5}\sum_{i=0}^{25}i=\frac{5\times{25+1}}{2}=65
\end{boldequation}
\end{itemize}
